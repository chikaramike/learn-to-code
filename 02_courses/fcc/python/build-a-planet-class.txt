class Planet():
    def __init__(self, name, planet_type, star):
        for value, label in [
            (name, "name"),
            (planet_type, "planet_type"),
            (star, "star"),
        ]:
            if not isinstance(value, str):
                raise TypeError(f"name, planet type, and star must be strings")
            if not value:
                raise ValueError(f"name, planet_type, and star must be non-empty strings")
        self.name = name
        self.planet_type = planet_type
        self.star = star

    def orbit(self):
        return f"{self.name} is orbiting around {self.star}..."

    def __str__(self):
        return f"Planet: {self.name} | Type: {self.planet_type} | Star: {self.star}"

planet_1 = Planet("mars", "small", "sun")
planet_2 = Planet("earth", "med", "sun")
planet_3 = Planet("saturn", "large", "sun")


print(planet_1)
print(planet_2)
print(planet_3)

print(planet_1.orbit())
print(planet_2.orbit())
print(planet_3.orbit())
