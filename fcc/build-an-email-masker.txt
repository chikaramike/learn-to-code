function maskEmail(email) {
    let atIndex = email.indexOf("@");
    if (atIndex === -1) return email; // return original if no "@"
    
    let masked = email.slice(0, 1) + "*".repeat(atIndex-2) + email.slice(atIndex -1, atIndex) + email.slice(atIndex);
    return masked;
  }
  
  let email = "email@email.com";
  console.log(maskEmail(email));
